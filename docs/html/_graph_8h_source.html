216/*
217 * Auxiliary function to remove an outgoing edge (with a given destination (d))
218 * from a vertex (this).
219 * Returns true if successful, and false if such edge does not exist.
220 */
221template<class T>
222bool Vertex<T>::removeEdge(T in) {
223 bool removedEdge = false;
224 auto it = adj.begin();
225 while (it != adj.end()) {
226 Edge<T> *edge = *it;
227 Vertex<T> *dest = edge->getDest();
228 if (dest->getCode() == in) {
229 it = adj.erase(it);
230 deleteEdge(edge);
231 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
232 } else {
233 it++;
234 }
235 }
236 return removedEdge;
237}

458/*
459 * Removes a vertex with a given content (in) from a graph (this), and
460 * all outgoing and incoming edges.
461 * Returns true if successful, and false if such vertex does not exist.
462 */
463template<class T>
464bool Graph<T>::removeVertex(const T &in) {
465 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
466 if ((*it)->getCode() == in) {
467 auto v = *it;
468 v->removeOutgoingEdges();
469 for (auto u: vertexSet) {
470 u->removeEdge(v->getCode());
471 }
472 vertexSet.erase(it);
473 delete v;
474 return true;
475 }
476 }
477 return false;
478}

524/*
525 * Performs a depth-first search (dfs) traversal in a graph (this).
526 * Returns a vector with the contents of the vertices by dfs order.
527 */
528template<class T>
529std::vector<T> Graph<T>::dfs() const {
530 std::vector<T> res;
531 for (auto v: vertexSet)
532 v->setVisited(false);
533 for (auto v: vertexSet)
534 if (!v->isVisited())
535 dfsVisit(v, res);
536 return res;
537}

539/*
540 * Performs a depth-first search (dfs) in a graph (this) from the source node.
541 * Returns a vector with the contents of the vertices by dfs order.
542 */
543template<class T>
544std::vector<T> Graph<T>::dfs(const T &source) const {
545 std::vector<int> res;
546 // Get the source vertex
547 auto s = findVertex(source);
548 if (s == nullptr) {
549 return res;
550 }
551 // Set that no vertex has been visited yet
552 for (auto v: vertexSet) {
553 v->setVisited(false);
554 }
555 // Perform the actual DFS using recursion
556 dfsVisit(s, res);
557
558 return res;
559}

561/*
562 * Auxiliary function that visits a vertex (v) and its adjacent, recursively.
563 * Updates a parameter with the list of visited node contents.
564 */
565template<class T>
566void Graph<T>::dfsVisit(Vertex<T> *v, std::vector<T> &res) const {
567 v->setVisited(true);
568 res.push_back(v->getCode());
569 for (auto &e: v->getAdj()) {
570 auto w = e->getDest();
571 if (!w->isVisited()) {
572 dfsVisit(w, res);
573 }
574 }
575}

577/****************** BFS ********************/
578/*
579 * Performs a breadth-first search (bfs) in a graph (this), starting
580 * from the vertex with the given source contents (source).
581 * Returns a vector with the contents of the vertices by bfs order.
582 */
583template<class T>
584std::vector<T> Graph<T>::bfs(const T &source) const {
585 std::vector<int> res;
586 // Get the source vertex
587 auto s = findVertex(source);
588 if (s == nullptr) {
589 return res;
590 }
591
592 // Set that no vertex has been visited yet
593 for (auto v: vertexSet) {
594 v->setVisited(false);
595 }
596
597 // Perform the actual BFS using a queue
598 std::queue<Vertex<T> *> q;
599 q.push(s);
600 s->setVisited(true);
601 while (!q.empty()) {
602 auto v = q.front();
603 q.pop();
604 res.push_back(v->getCode());
605 for (auto &e: v->getAdj()) {
606 auto w = e->getDest();
607 if (!w->isVisited()) {
608 q.push(w);
609 w->setVisited(true);
610 }
611 }
612 }
613 return res;
614}

616/****************** isDAG ********************/
617/*
618 * Performs a depth-first search in a graph (this), to determine if the graph
619 * is acyclic (acyclic directed graph or DAG).
620 * During the search, a cycle is found if an edge connects to a vertex
621 * that is being processed in the stack of recursive calls (see theoretical classes).
622 * Returns true if the graph is acyclic, and false otherwise.
623 */
624
625template<class T>
626bool Graph<T>::isDAG() const {
627 for (auto v: vertexSet) {
628 v->setVisited(false);
629 v->setProcesssing(false);
630 }
631 for (auto v: vertexSet) {
632 if (!v->isVisited()) {
633 if (!dfsIsDAG(v)) return false;
634 }
635 }
636 return true;
637}
638
643template<class T>
644bool Graph<T>::dfsIsDAG(Vertex<T> *v) const {
645 v->setVisited(true);
646 v->setProcesssing(true);
647 for (auto e: v->getAdj()) {
648 auto w = e->getDest();
649 if (w->isProcessing()) return false;
650 if (!w->isVisited()) {
651 if (!dfsIsDAG(w)) return false;
652 }
653 }
654 v->setProcesssing(false);
655 return true;
656}

663/*
664 * Performs a topological sorting of the vertices of a graph (this).
665 * Returns a vector with the contents of the vertices by topological order.
666 * If the graph has cycles, returns an empty vector.
667 * Follows the algorithm described in theoretical classes.

670template<class T>
671std::vector<T> Graph<T>::topsort() const {
672 std::vector<int> res;
673
674 for (auto v: vertexSet) {
675 v->setIndegree(0);
676 }
677 for (auto v: vertexSet) {
678 for (auto e: v->getAdj()) {
679 unsigned int indegree = e->getDest()->getIndegree();
680 e->getDest()->setIndegree(indegree + 1);
681 }
682 }
683
684 std::queue<Vertex<T> *> q;
685 for (auto v: vertexSet) {
686 if (v->getIndegree() == 0) {
687 q.push(v);
688 }
689 }
690
691 while (!q.empty()) {
692 Vertex<T> *v = q.front();
693 q.pop();
694 res.push_back(v->getCode());
695 for (auto e: v->getAdj()) {
696 auto w = e->getDest();
697 w->setIndegree(w->getIndegree() - 1);
698 if (w->getIndegree() == 0) {
699 q.push(w);
700 }
701 }
702 }
703
704 if (res.size() != vertexSet.size()) {
705 //std::cout << "Impossible topological ordering!" << std::endl;
706 res.clear();
707 return res;
708 }
709
710 return res;
711}
